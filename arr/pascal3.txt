// Printing the whole pascal's triangle...
#include<iostream>
using namespace std;
int find(int n,int r){
    int res = 1;
 for (int i = 0; i < r; i++)
 {
  res = res * (n-i);
  res = res / (i+1);
 }
 return res;
}
void result(int n){
    int ans;
    cout<<"The pascal triangle is: "<<endl;
    for (int i = 1; i <= n; i++)
    {
       for (int j = 0; j < i; j++)
       {
         ans = find(i-1,j);
         cout<<ans<<" ";
       }
       cout<<endl;
    }
}
int main()
{
    int row;
    cout<<"Enter the row no: ";
    cin>>row;
    result(row);
    return 0;
}
