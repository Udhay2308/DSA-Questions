// Moving zeroes to end of the array...
#include<iostream>
#include<vector>
using namespace std;

void printarray(vector<int> arr,int n){
    for (int i = 0; i < n ; i++)
    {
       cout<<arr[i]<<" ";
    }
}
void move(vector<int>&arr,int n){
   int j = -1;
   for (int i = 0; i < n; i++)
   {
     if (arr[i] == 0)
     {
       j = i;
       break;
     }
   }
   for (int i = j + 1; i < n; i++)
   {
     if (arr[i] != 0)
     {
       swap(arr[i],arr[j]);
       j++;
     }
   }
}
int main()
{
    int n;
    cout<<"Enter the size of the array: ";
    cin>>n;
    vector<int> arr(n);
    for (int i = 0; i < n; i++)
    {
      cout<<"Enter the "<<i<<"th element of the array: ";
      cin>>arr[i];
    }
    cout<<"Given array is: ";
    for (int i = 0; i < n ; i++)
    {
       cout<<arr[i]<<" ";
    }
    cout<<endl;
    move(arr,n);
    cout<<"Array after moving zeroes to the end: ";
    printarray(arr,n);
    cout<<endl;
    return 0;
}
