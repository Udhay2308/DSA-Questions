// Finding the next lexicographic order...
#include<bits/stdc++.h>
using namespace std;

    
 vector<int> next(vector<int>& arr) {
       int index = -1;
       int n = arr.size();
       for (int i = n - 2; i >= 0; i--)
       {
          if(arr[i] < arr[i + 1]){
            index = i;
            break;
          }
       }
       if(index == -1){
            reverse(arr.begin(),arr.end());
            cout<<"hi";
            return arr;
       }
       for (int i = n - 1; i > index; i--)
       {
          if(arr[i] > arr[index]){
            swap(arr[i],arr[index]);
            break;
          }
       }
       reverse(arr.begin() + index + 1,arr.end());
       return arr;
    }
    int main()
    {
        vector<int>arr = {1,3,2};
        vector<int> ans = next(arr);
        cout<<"The next permutation is: "<<endl;
        for (int i = 0; i < ans.size(); i++)
        {
            cout<<ans[i]<<" ";
        }
        
        return 0;
    }
