#include <bits/stdc++.h>
using namespace std;
void find(vector<string>&ans,string s,int n){
if(s.length() == n){
ans.push_back(s);
return;
}
s.push_back('0');
find(ans,s,n);
s.pop_back();
s.push_back('1');
find(ans,s,n);
}
int main()
{   int n = 3;
    vector<string>ans;
    string s;
    find(ans,s,n);
    for (const std::string& word : ans) {
        std::cout << word << std::endl;
    }
    return 0;
}
