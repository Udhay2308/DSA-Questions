// Recursive selection sort..
#include<iostream>
#include<climits>
using namespace std;

void printarray(int arr[],int n){
    for (int i = 0; i < n ; i++)
    {
       cout<<arr[i]<<" ";
    }
}
void selection_sort(int arr[],int n,int a){
    int min = a;
    if (a > n-1)
    {
     return;
    }
        for (int i = a; i < n; i++)
    {
        if (arr[i] < arr[min])
        {
            min = i;
        }
    }
    int temp = arr[a];
    arr[a] = arr[min];
    arr[min] = temp;
    selection_sort(arr,n,a+1);
}
int main()
{  
    int n;
    int a = 0;
    cout<<"Enter the size of the array: ";
    cin>>n;
    int* arr = new int[n];
    for (int i = 0; i < n; i++)
    {
      cout<<"Enter the "<<i<<"th element of the array: ";
      cin>>arr[i];
    }
    cout<<"The original array is: "<<endl;
    printarray(arr,n);
    cout<<endl;
    selection_sort(arr,n,a);
    cout<<"The sorted array is: "<<endl;
    printarray(arr,n);
    return 0;
}
