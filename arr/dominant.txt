// Finding the majority element of the array if exists...
#include <bits/stdc++.h>
using namespace std;

int dominance(vector<int>arr,int n){
    int i = 0;
    int j = 0;
    int max_occ = 0;
    sort(arr.begin(),arr.end());
    int count = 0;
    while (j < n)
    {
       if(arr[i] == arr[j]){
        count++;
        j++;
       }
       else if(arr[j] > arr[i] || j == n-1){
        max_occ = max(max_occ,count);
        count = 0;
        i = j;
       }
    }
    return max_occ;
}
int main()
{
    int n = 10;
    vector<int>arr = {2,2,1,1,1,2,2,2,2,2};
    int ans =  dominance(arr,n);
    cout<<"The dominant element in this array is "<<ans;
    return 0;
}
