// Finding the maximum sum that a subarray can contain..
#include <bits/stdc++.h>
using namespace std;

int find(int arr[],int n){
     int max_sum = INT_MIN;
     int sum = 0;
     for (int i = 0; i < n; i++)
    {   
       sum = sum + arr[i];

       if(sum > max_sum){
        max_sum = sum;
       }

       if(sum < 0){
        sum = 0;
       }
    }
    return max_sum;
}
int main()
{
    int n = 8;
   int arr[] = {-2,-3,4,-1,-2,1,5,-3};
   int ans =  find(arr,n);
   cout<<"Maximum sum is: "<<ans;
    return 0;
}
