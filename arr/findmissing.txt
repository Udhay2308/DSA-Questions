// Find missing element...
#include<iostream>
using namespace std;

int find(int arr[],int n){
    for (int i = 1; i <= n; i++)
    {  
       int count = 0;
       for (int j = 0; j < n-1; j++)
       {  
          if (i == arr[j])
          {
            count = 1;
            break;
          }
       }
       if (count == 0)
       {
        return i;
       }
    }
       return -1;
}
int main()
{
    int n;
    cout<<"Enter the value of the n: ";
    cin>>n;
    int* arr = new int[n];
    cout<<"Enter the elements of array between 1 to n"<<endl;
    for (int i = 0; i < n-1; i++)
    {
      cout<<"Enter the "<<i<<"th element of the array: ";
      cin>>arr[i];
    }
    cout<<"Given array is: ";
    for (int i = 0; i < n -1; i++)
    {
       cout<<arr[i]<<" ";
    }
    cout<<endl;
    int ans = find(arr,n);
    cout<<"Missing element is: "<<ans;
    return 0;
}
