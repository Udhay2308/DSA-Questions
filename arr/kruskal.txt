#include <stdio.h>

#define MAX 30

int parent[MAX];

int find(int i) {
    while (parent[i] != i)
        i = parent[i];
    return i;
}

void unionSets(int i, int j) {
    int a = find(i);
    int b = find(j);
    parent[a] = b;
}

int main() {
    int n = 4; // number of vertices
    int cost[4][4] = {
        {999, 10, 6, 5},
        {10, 999, 999, 15},
        {6, 999, 999, 4},
        {5, 15, 4, 999}
    };

    for (int i = 0; i < n; i++)
        parent[i] = i;

    int edges = 0, mincost = 0;

    while (edges < n - 1) {
        int min = 999, a = -1, b = -1;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (find(i) != find(j) && cost[i][j] < min) {
                    min = cost[i][j];
                    a = i;
                    b = j;
                }
            }
        }
        unionSets(a, b);
        printf("Edge %d: (%d - %d) cost: %d\n", edges++, a, b, min);
        mincost += min;
    }

    printf("Minimum cost = %d\n", mincost);
    return 0;
}
