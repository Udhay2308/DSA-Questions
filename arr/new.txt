// Recursive bubble sort...
#include<iostream>
#include<climits>
using namespace std;

void printarray(int arr[],int n){
    for (int i = 0; i < n ; i++)
    {
       cout<<arr[i]<<" ";
    }
}
void bubble_sort(int arr[],int i,int n){
   
    if (i > n-1)
    {
       return;
    }
    for (int j = 0; j < n-i-1; j++)
    { 
         if (arr[j] > arr[j+1])
    {
        int temp = arr[j];
        arr[j] = arr[j+1];
        arr[j+1] = temp;
    }
    }
    bubble_sort(arr,i+1,n);
    
}
int main()
{  
    int n;
    int i = 0;
    cout<<"Enter the size of the array: ";
    cin>>n;
    int* arr = new int[n];
    for (int i = 0; i < n; i++)
    {
      cout<<"Enter the "<<i<<"th element of the array: ";
      cin>>arr[i];
    }
    cout<<"The original array is: "<<endl;
    printarray(arr,n);
    cout<<endl;
    bubble_sort(arr,i,n);
    cout<<"The sorted array is: "<<endl;
    printarray(arr,n);
    return 0;
}
