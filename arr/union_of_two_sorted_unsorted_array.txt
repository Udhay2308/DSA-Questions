vector<int> common_elements(vector<int> &common, vector<int> &arr1, vector<int> &arr2)
{
    int size1 = arr1.size();
    int size2 = arr2.size();
    int size = max(size1, size2);
    for (int i = 0; i < size1; i++)
    {
        if (arr1[i] == arr1[i - 1])
        {
            continue;
        }

            for (int j = 0; j < size2; j++)
            {
                if (arr1[i] == arr2[j])
                {
                    common.push_back(arr1[i]);
                    break;
                }
            }
    }
    cout << "Common Array is: ";
    for (int i = 0; i < common.size(); i++)
    {
        cout << common[i] << " ";
    }
    return common;
}
vector<int> unique1(vector<int>arr1,vector<int>&arr1_unique,vector<int>common){
    for (int i = 0; i < arr1.size(); i++)
    {  int count = 0;
       for (int j = 0; j < common.size(); j++)
       {
         if (arr1[i] != common[j])
         {
           count++;
         }
         else{
         break;
         }
       }
       if (count == common.size())
       {
        arr1_unique.push_back(arr1[i]);
       }  
    }
    cout << "Unique arr1 is: ";
    for (int i = 0; i < arr1_unique.size(); i++)
    {
        cout << arr1_unique[i] << " ";
    }
    return arr1_unique;
}
