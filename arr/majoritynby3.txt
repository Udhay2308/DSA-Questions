// Majority element.occur more than n/3 times...
#include<iostream>
#include<vector>
#include<map>
using namespace std;
vector<int> majority(vector<int> arr){
    int n = arr.size();
    map<int,int>mp;
    vector<int>ans;
    for (int i = 0; i < n; i++)
    {
       mp[arr[i]]++;
       if (mp[arr[i]] > (n/3))
       {
         ans.push_back(arr[i]);
       }
       if (ans.size() == 2)
       {
        break;
       }
    }
    return ans;
}
int main()
{  
    vector<int>arr = {11,33,33,11,11,33};
    vector<int>ans = majority(arr);
    cout<<"The element which occurs more than n/3 times are: ";
    for (auto it : ans) {
        cout << it << " "; 
    }
    return 0;
}
