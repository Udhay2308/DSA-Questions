// Recursive insertion sort...
#include<iostream>
#include<climits>
using namespace std;

void printarray(int arr[],int n){
    for (int i = 0; i < n ; i++)
    {
       cout<<arr[i]<<" ";
    }
}
void insertion_sort(int arr[],int i,int n){
   if (i > n-1)
   {
    return;
   }
       for (int j = i-1; j >= 0; j--)
       {
         if (arr[i] < arr[j])
         {
            int temp = arr[i];
            arr[i] = arr[j];
            arr[j] = temp;
            i--;
         }
       }
       insertion_sort(arr,i+1,n);
    }
int main()
{  
    int n;
    int i = 1;
    cout<<"Enter the size of the array: ";
    cin>>n;
    int* arr = new int[n];
    for (int i = 0; i < n; i++)
    {
      cout<<"Enter the "<<i<<"th element of the array: ";
      cin>>arr[i];
    }
    cout<<"The original array is: "<<endl;
    printarray(arr,n);
    cout<<endl;
    insertion_sort(arr,i,n);
    cout<<"The sorted array is: "<<endl;
    printarray(arr,n);
    return 0;
}
