// To find the leaders in the given array..
#include<bits/stdc++.h>
using namespace std;

vector<int> lead(int arr[],int n){
    vector<int>ans;
  for (int i = 0; i < n; i++)
  { 
     bool leader = true;
    for (int j = i + 1; j < n; j++)
    {
        if(arr[j] > arr[i]){
            leader = false;
        }
    }
    if(leader){
        ans.push_back(arr[i]);
    }
  }
  return ans;
}
int main()
{
    int n = 6;
    int arr[] = {10,22,12,3,0,6};
    vector<int>ans = lead(arr,n);
    cout<<"The leaders in the given array is: "<<endl;
    for (int i = 0; i < ans.size(); i++)
    {
       cout<<ans[i]<<" ";
    }
    return 0;
}
