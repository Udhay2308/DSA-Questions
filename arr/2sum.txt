// Checking if the pair with given sum is present in array or not..
#include<iostream>
#include<map>
using namespace std;

bool sum2(int arr[],int n,int target){
  map<int,int>mp;
  for (int i = 0; i < n; i++)
  {
    if (mp.find(target - arr[i]) != mp.end())
    {
      return true;
    }
    mp[arr[i]] = i;
  }
  return false;
}
int main()
{   int n = 5;
    int arr[] = {2,6,5,8,11};
    int target;
    cout<<"Enter the target(sum): ";
    cin>>target;
    bool ans = sum2(arr,n,target);
    if (ans)
    {
      cout<<"Yes";
    }
    else{
        cout<<"NO";
    }
    return 0;
}
