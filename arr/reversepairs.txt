// To find the missing and the repeating number..
#include<iostream>
#include<vector>
#include<climits>
#include<map>
using namespace std;
int rev_pair(vector<int> arr){
  int n = arr.size();
    int count = 0;
  for (int i = n - 1; i > 0; i--)
  {
   for (int j = 0; j < i; j++)
   {
    if (arr[j] > (2*arr[i]))
    {
      count++;
    }
   }
  }
   return count;
}
int main()
{  
    vector<int>arr = {4, 1, 2, 3, 1};
    int ans = rev_pair(arr);
    cout<<"The number of reverse pairs in the given array is: "<<ans;
    return 0;
}
