// Rotate left by k position.
#include<iostream>
using namespace std;

void reverse(int arr[],int start,int end){
    while (start <= end){
       int temp = arr[start];
       arr[start] = arr[end];
       arr[end] = temp;
       start++;
       end--;
    }
}
int rotate_left(int arr[],int n,int k){
    reverse(arr,0,k - 1);
    reverse(arr,k,n - 1);
    reverse(arr,0,n - 1);
}
int main()
{
    int n;
    cout<<"Enter the size of the array: ";
    cin>>n;
    int* arr = new int[n];
    for (int i = 0; i < n; i++)
    {
      cout<<"Enter the "<<i<<"th element of the array: ";
      cin>>arr[i];
    }
    cout<<"Given array is: ";
    for (int i = 0; i < n ; i++)
    {
       cout<<arr[i]<<" ";
    }
    cout<<endl;
    int k;
    cout<<"Enter the number by which you want to rotate the array in left direction: ";
    cin>>k;
    rotate_left(arr,n,k);
    cout<<"Array after left rotation by k position is: ";
    for (int i = 0; i < n ; i++)
    {
       cout<<arr[i]<<" ";
    }
    return 0;
}
