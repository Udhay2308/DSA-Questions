// Subarray with sum zero...
#include <bits/stdc++.h>
using namespace std;

int solve(vector<int>& arr) {
    int maxLen = 0;
    unordered_map<int, int> presum;
    int sum = 0;
    for (int i = 0; i < arr.size(); i++)
    {
       sum += arr[i];
       if (sum == 0) maxLen = i + 1;
       else if(presum.find(sum) != presum.end()){
            maxLen = max(maxLen,i - presum[sum]);
       }
       else{
        presum[sum] = i;
       }
    }
    return maxLen;
}

int main() {
    vector<int> arr = {9, -3, 3, -1, 6, -5};
    cout<<"The largest subarray with sum zero is: "<<solve(arr) << endl;
    return 0;
}
