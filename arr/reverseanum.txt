// Reversing the number
#include <iostream>
using namespace std;
int main()
{
    int n;
    cout << ("Enter the value of n: ");
    cin >> n;
    int remainder;
    int result = 0;
    while (n > 0)
    {
        remainder = n % 10;
        n = n / 10;
        result = result * 10 + remainder;
    }
    cout <<"The reversed number is: "<< result;
    return 0;
}
